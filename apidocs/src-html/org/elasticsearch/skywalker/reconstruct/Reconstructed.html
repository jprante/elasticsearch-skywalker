019    package org.elasticsearch.skywalker.reconstruct;
020    
021    import java.util.ArrayList;
022    import java.util.Collections;
023    import java.util.HashMap;
024    import java.util.HashSet;
025    import java.util.List;
026    import java.util.Map;
027    import org.apache.lucene.document.Fieldable;
028    
029    public class Reconstructed {
030    
031        private Map<String, Fieldable[]> storedFields;
032        private Map<String, GrowableStringArray> reconstructedFields;
033    
034        public Reconstructed() {
035            storedFields = new HashMap();
036            reconstructedFields = new HashMap();
037        }
038    
039        /**
040         * Construct an instance of this class using existing field data.
041         *
042         * @param storedFields field data of stored fields
043         * @param reconstructedFields field data of unstored fields
044         */
045        public Reconstructed(Map<String, Fieldable[]> storedFields,
046                Map<String, GrowableStringArray> reconstructedFields) {
047            this.storedFields = storedFields;
048            this.reconstructedFields = reconstructedFields;
049        }
050    
051        /**
052         * Get an alphabetically sorted list of field names.
053         */
054        public List<String> getFieldNames() {
055            HashSet<String> names = new HashSet();
056            names.addAll(storedFields.keySet());
057            names.addAll(reconstructedFields.keySet());
058            ArrayList<String> res = new ArrayList(names.size());
059            res.addAll(names);
060            Collections.sort(res);
061            return res;
062        }
063    
064        public boolean hasField(String name) {
065            return storedFields.containsKey(name) || reconstructedFields.containsKey(name);
066        }
067    
068        /**
069         * @return the storedFields
070         */
071        public Map<String, Fieldable[]> getStoredFields() {
072            return storedFields;
073        }
074    
075        /**
076         * @return the reconstructedFields
077         */
078        public Map<String, GrowableStringArray> getReconstructedFields() {
079            return reconstructedFields;
080        }
081    }
