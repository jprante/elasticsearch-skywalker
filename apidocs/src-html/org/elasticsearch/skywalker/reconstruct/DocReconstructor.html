021    import java.util.Arrays;
022    import java.util.HashSet;
023    import java.util.Iterator;

026    import org.apache.lucene.document.Document;
027    import org.apache.lucene.document.Fieldable;
028    import org.apache.lucene.index.FieldInfo;
029    import org.apache.lucene.index.IndexReader;

032    import org.apache.lucene.index.TermFreqVector;

038    /**
039     * This class attempts to reconstruct all fields from a document existing in a
040     * Lucene index. This operation may be (and usually) is lossy - e.g. unstored
041     * fields are rebuilt from terms present in the index, and these terms may have
042     * been changed (e.g. lowercased, stemmed), and many other input tokens may have
043     * been skipped altogether by the Analyzer, when fields were originally added to
044     * the index.
045     *
046     * @author ab
047     *
048     */
049    public class DocReconstructor {
050    
051        //private ProgressNotification progress = new ProgressNotification();
052        private String[] fieldNames = null;
053        private IndexReader reader = null;
054    
055        /**
056         * Prepare a document reconstructor.
057         *
058         * @param reader IndexReader to read from.
059         * @throws Exception
060         */
061        public DocReconstructor(IndexReader reader) throws Exception {
062            this(reader, null, -1);
063        }
064    
065        /**
066         * Prepare a document reconstructor.
067         *
068         * @param reader IndexReader to read from.
069         * @param fieldNames if non-null or not empty, data will be collected only
070         * from these fields, otherwise data will be collected from all fields
071         * @param numTerms total number of terms in the index, or -1 if unknown
072         * (will be calculated)
073         * @throws Exception
074         */
075        public DocReconstructor(IndexReader reader, String[] fieldNames, int numTerms) throws Exception {
076            if (reader == null) {
077                throw new Exception("IndexReader cannot be null.");
078            }
079            this.reader = reader;
080            if (fieldNames == null || fieldNames.length == 0) {
081                // collect fieldNames
082                final Iterator<FieldInfo> it = ReaderUtil.getMergedFieldInfos(reader).iterator();
083                final List<String> names = new LinkedList();
084                while (it.hasNext()) {
085                    final FieldInfo info = it.next();
086                    names.add(info.name);
087                }
088                this.fieldNames = names.toArray(new String[names.size()]);
089            } else {
090                this.fieldNames = fieldNames;
091            }
092            if (numTerms == -1) {
093                TermEnum te = null;
094                try  {
095                    te = reader.terms();
096                    numTerms = 0;
097                    while (te.next()) {
098                        numTerms++;
099                    }
100                } finally {
101                    te.close();
102                }
103            }
104        }
105    
106        /**
107         * Reconstruct document fields.
108         *
109         * @param docNum document number. If this document is deleted, but the index
110         * is not optimized yet, the reconstruction process may still yield the
111         * reconstructed field content even from deleted documents.
112         * @return reconstructed document
113         * @throws Exception
114         */
115        public Reconstructed reconstruct(int docNum) throws Exception {
116            if (docNum < 0 || docNum > reader.maxDoc()) {
117                throw new Exception("Document number outside of valid range.");
118            }
119            Reconstructed res = new Reconstructed();
120            if (reader.isDeleted(docNum)) {
121                throw new Exception("Document is deleted.");
122            } else {
123                Document doc = reader.document(docNum);
124                for (int i = 0; i < fieldNames.length; i++) {
125                    Fieldable[] fs = doc.getFieldables(fieldNames[i]);
126                    if (fs != null && fs.length > 0) {
127                        res.getStoredFields().put(fieldNames[i], fs);
128                    }
129                }
130            }
131            // collect values from unstored fields
132            HashSet<String> fields = new HashSet(Arrays.asList(fieldNames));
133            // try to use term vectors if available
134            //progress.maxValue = fieldNames.length;
135            //progress.curValue = 0;
136            //progress.minValue = 0;
137            for (int i = 0; i < fieldNames.length; i++) {
138                TermFreqVector tvf = reader.getTermFreqVector(docNum, fieldNames[i]);
139                if (tvf != null && tvf.size() > 0 && (tvf instanceof TermPositionVector)) {
140                    TermPositionVector tpv = (TermPositionVector) tvf;
141                    //progress.message = "Reading term vectors ...";
142                    //progress.curValue = i;
143                    //setChanged();
144                    //notifyObservers(progress);
145                    String[] tv = tpv.getTerms();
146                    for (int k = 0; k < tv.length; k++) {
147                        // do we have positions?
148                        int[] posArr = tpv.getTermPositions(k);
149                        if (posArr == null) {
150                            // only offsets
151                            TermVectorOffsetInfo[] offsets = tpv.getOffsets(k);
152                            if (offsets.length == 0) {
153                                continue;
154                            }
155                            // convert offsets into positions
156                            posArr = convertOffsets(offsets);
157                        }
158                        GrowableStringArray gsa = res.getReconstructedFields().get(fieldNames[i]);
159                        if (gsa == null) {
160                            gsa = new GrowableStringArray();
161                            res.getReconstructedFields().put(fieldNames[i], gsa);
162                        }
163                        for (int m = 0; m < posArr.length; m++) {
164                            gsa.append(posArr[m], "|", tv[k]);
165                        }
166                    }
167                    fields.remove(fieldNames[i]); // got what we wanted
168                }
169            }
170            String term;
171            TermPositions tp = null;
172            try {
173                tp = reader.termPositions();
174                for (String fld : fields) {
175                    TermEnum te = null;
176                    try {
177                        te = reader.terms(new Term(fld, ""));
178                        if (te == null || te.term() == null || !te.term().field().equals(fld)) {
179                            continue;
180                        }
181                        // TermEnum is already positioned
182                        do {
183                            if (!te.term().field().equals(fld)) {
184                                // end of terms in this field
185                                break;
186                            }
187                            tp.seek(te.term());
188                            if (!tp.skipTo(docNum) || tp.doc() != docNum) {
189                                // this term is not found in the doc
190                                continue;
191                            }
192                            term = te.term().text();
193                            GrowableStringArray gsa = (GrowableStringArray) res.getReconstructedFields().get(te.term().field());
194                            if (gsa == null) {
195                                gsa = new GrowableStringArray();
196                                res.getReconstructedFields().put(te.term().field(), gsa);
197                            }
198                            for (int k = 0; k < tp.freq(); k++) {
199                                int pos = tp.nextPosition();
200                                gsa.append(pos, "|", term);
201                            }
202                        } while (te.next());
203                    } finally {
204                        te.close();
205                    }
206                }
207            } finally {
208                tp.close();
209            }
210            //progress.message = "Done.";
211            //progress.curValue = 100;
212            //setChanged();
213            //notifyObservers(progress);
214            return res;
215        }
216    
217        private int[] convertOffsets(TermVectorOffsetInfo[] offsets) {
218            int[] posArr = new int[offsets.length];
219            int curPos = 0;
220            int maxDelta = 3; // allow 3 characters diff, otherwise insert a skip
221            int avgTermLen = 5; // assume this is the avg. term length of missing terms
222            for (int m = 0; m < offsets.length; m++) {
223                int curStart = offsets[m].getStartOffset();
224                if (m > 0) {
225                    int prevEnd = offsets[m - 1].getEndOffset();
226                    int prevStart = offsets[m - 1].getStartOffset();
227                    if (curStart == prevStart) {
228                        curPos--; // overlapping token
229                    } else {
230                        if (prevEnd + maxDelta < curStart) { // possibly a gap
231                            // calculate the number of missing tokens
232                            int increment = (curStart - prevEnd) / (maxDelta + avgTermLen);
233                            if (increment == 0) {
234                                increment++;
235                            }
236                            curPos += increment;
237                        }
238                    }
239                }
240                posArr[m] = curPos;
241                curPos++;
242            }
243            return posArr;
244        }
