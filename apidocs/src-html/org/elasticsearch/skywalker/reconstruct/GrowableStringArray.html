019    package org.elasticsearch.skywalker.reconstruct;
020    
021    /**
022     * Simple Vector-like implementation of a growable String array.
023     *
024     */
025    public class GrowableStringArray {
026    
027        public int INITIAL_SIZE = 20;
028        private int size = 0;
029        private String[] array = null;
030    
031        public int size() {
032            return size;
033        }
034    
035        /**
036         * Sets the value at specified index. If index is outside range the array is
037         * automatically expanded.
038         *
039         * @param index where to set the value
040         * @param value
041         */
042        public void set(int index, String value) {
043            if (array == null) {
044                array = new String[INITIAL_SIZE];
045            }
046            if (array.length < index + 1) {
047                String[] newArray = new String[index + INITIAL_SIZE];
048                System.arraycopy(array, 0, newArray, 0, array.length);
049                array = newArray;
050            }
051            if (index > size - 1) {
052                size = index + 1;
053            }
054            array[index] = value;
055        }
056    
057        /**
058         * Appends the separator and value at specified index. If no value exists at
059         * the specified position, this is equivalent to {@link #set(int, String)} -
060         * no separator is appended in that case.
061         *
062         * @param index selected position
063         * @param sep separator
064         * @param value value
065         */
066        public void append(int index, String sep, String value) {
067            String oldVal = get(index);
068            if (oldVal == null) {
069                set(index, value);
070            } else {
071                set(index, oldVal + sep + value);
072            }
073        }
074    
075        /**
076         * Return the value at specified index.
077         *
078         * @param index
079         * @return the value
080         */
081        public String get(int index) {
082            if (array == null || index < 0 || index > array.length - 1) {
083                return null;
084            }
085            return array[index];
086        }
087    
088        public String toString(String separator) {
089            StringBuilder sb = new StringBuilder();
090            String sNull = "null";
091            int k = 0, m = 0;
092            for (int j = 0; j < size(); j++) {
093                if (get(j) == null) {
094                    k++;
095                } else {
096                    if (sb.length() > 0) {
097                        sb.append(separator);
098                    }
099                    if (m > 0 && m % 5 == 0) {
100                        sb.append('\n');
101                    }
102                    if (k > 0) {
103                        sb.append(sNull).append("_").append(k).append(separator);
104                        k = 0;
105                        m++;
106                    }
107                    sb.append(get(j));
108                    m++;
109                }
110            }
111            return sb.toString();
112        }
113    }
