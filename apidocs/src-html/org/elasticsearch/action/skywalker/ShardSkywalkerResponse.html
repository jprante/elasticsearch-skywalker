019    package org.elasticsearch.action.skywalker;
020    
021    import java.io.IOException;
022    import java.util.Map;
023    import org.elasticsearch.action.support.broadcast.BroadcastShardOperationResponse;
024    import org.elasticsearch.common.io.stream.StreamInput;
025    import org.elasticsearch.common.io.stream.StreamOutput;
026    
027    public class ShardSkywalkerResponse extends BroadcastShardOperationResponse {
028    
029        private Map<String, Object> response;
030    
031        ShardSkywalkerResponse() {
032        }
033    
034        public ShardSkywalkerResponse(String index, int shardId) {
035            super(index, shardId);
036        }
037    
038        public ShardSkywalkerResponse setResponse(Map<String, Object> response) {
039            this.response = response;
040            return this;
041        }
042    
043        public Map<String, Object> getResponse() {
044            return response;
045        }
046    
047        @Override
048        public void readFrom(StreamInput in) throws IOException {
049            super.readFrom(in);
050            Map<String, Object> fields = in.readMap();
051        }
052    
053        @Override
054        public void writeTo(StreamOutput out) throws IOException {
055            super.writeTo(out);
056            out.writeMap(response);
057        }
058    }
