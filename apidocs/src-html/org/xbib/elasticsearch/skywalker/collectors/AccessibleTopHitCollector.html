019package org.xbib.elasticsearch.skywalker.collectors;
020
021import org.apache.lucene.index.AtomicReaderContext;
022import org.apache.lucene.search.Scorer;
023import org.apache.lucene.search.TopDocs;
024import org.apache.lucene.search.TopScoreDocCollector;
025
026import java.io.IOException;
027
028/**
029 * Accessible top hit collector
030 *
031 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
032 */
033public class AccessibleTopHitCollector extends AccessibleHitCollector {
034
035    private TopScoreDocCollector tdc;
036    private TopDocs topDocs = null;
037    private int size;
038
039    public AccessibleTopHitCollector(int size, boolean outOfOrder, boolean shouldScore) {
040        tdc = TopScoreDocCollector.create(size, outOfOrder);
041        this.shouldScore = shouldScore;
042        this.outOfOrder = outOfOrder;
043        this.size = size;
044    }
045
046    @Override
047    public int getDocId(int pos) {
048        if (topDocs == null) {
049            topDocs = tdc.topDocs();
050        }
051        return topDocs.scoreDocs[pos].doc;
052    }
053
054    @Override
055    public float getScore(int pos) {
056        if (topDocs == null) {
057            topDocs = tdc.topDocs();
058        }
059        return topDocs.scoreDocs[pos].score;
060    }
061
062    @Override
063    public int getTotalHits() {
064        return tdc.getTotalHits();
065    }
066
067    @Override
068    public boolean acceptsDocsOutOfOrder() {
069        return tdc.acceptsDocsOutOfOrder();
070    }
071
072    @Override
073    public void collect(int doc) throws IOException {
074        tdc.collect(doc);
075    }
076
077    @Override
078    public void setNextReader(AtomicReaderContext context) throws IOException {
079        this.docBase = context.docBase;
080        tdc.setNextReader(context);
081    }
082
083    @Override
084    public void setScorer(Scorer scorer) throws IOException {
085        if (shouldScore) {
086            tdc.setScorer(scorer);
087        } else {
088            tdc.setScorer(NoScoringScorer.INSTANCE);
089        }
090    }
091
092    @Override
093    public void reset() {
094        tdc = TopScoreDocCollector.create(size, outOfOrder);
095        topDocs = null;
096    }
097}
