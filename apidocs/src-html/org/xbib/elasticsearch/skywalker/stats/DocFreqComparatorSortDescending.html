002    package org.xbib.elasticsearch.skywalker.stats;
003    
004    import java.util.Comparator;
005    
006    /**
007     *  Document frequence comparator for descending sort
008     */
009    public class DocFreqComparatorSortDescending implements Comparator<TermStats> {
010    
011        public int compare(TermStats a, TermStats b) {
012            if (a.docFreq() < b.docFreq()) {
013                return 1;
014            } else if (a.docFreq() > b.docFreq()) {
015                return -1;
016            } else {
017                return 0;
018            }
019        }
020    }
