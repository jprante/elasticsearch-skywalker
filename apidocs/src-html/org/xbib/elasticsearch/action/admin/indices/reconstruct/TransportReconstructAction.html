001    package org.xbib.elasticsearch.action.admin.indices.reconstruct;
002    
003    import org.apache.lucene.index.IndexReader;
004    import org.elasticsearch.ElasticsearchException;
005    import org.elasticsearch.action.ShardOperationFailedException;
006    import org.elasticsearch.action.support.DefaultShardOperationFailedException;
007    import org.elasticsearch.action.support.broadcast.BroadcastShardOperationFailedException;
008    import org.elasticsearch.action.support.broadcast.BroadcastShardOperationResponse;
009    import org.elasticsearch.action.support.broadcast.TransportBroadcastOperationAction;
010    import org.elasticsearch.cluster.ClusterService;
011    import org.elasticsearch.cluster.ClusterState;
012    import org.elasticsearch.cluster.block.ClusterBlockException;
013    import org.elasticsearch.cluster.block.ClusterBlockLevel;
014    import org.elasticsearch.cluster.routing.GroupShardsIterator;
015    import org.elasticsearch.cluster.routing.ShardRouting;
016    import org.elasticsearch.common.inject.Inject;
017    import org.elasticsearch.common.settings.Settings;
018    import org.elasticsearch.index.engine.Engine;
019    import org.elasticsearch.index.service.IndexService;
020    import org.elasticsearch.index.shard.service.InternalIndexShard;
021    import org.elasticsearch.indices.IndicesService;
022    import org.xbib.elasticsearch.skywalker.reconstruct.DocumentReconstructor;
023    import org.elasticsearch.threadpool.ThreadPool;
024    import org.elasticsearch.transport.TransportService;
025    
026    import java.io.IOException;
027    import java.util.List;
028    import java.util.concurrent.atomic.AtomicReferenceArray;
029    
030    import static org.elasticsearch.common.collect.Lists.newArrayList;
031    
032    /**
033     *  Transport reconstruct index action
034     */
035    public class TransportReconstructAction extends TransportBroadcastOperationAction<ReconstructIndexRequest, ReconstructIndexResponse, ShardReconstructIndexRequest, BroadcastShardOperationResponse> {
036    
037        private final IndicesService indicesService;
038    
039        @Inject
040        public TransportReconstructAction(Settings settings, ThreadPool threadPool, ClusterService clusterService,
041                                          TransportService transportService, IndicesService indicesService) {
042            super(settings, threadPool, clusterService, transportService);
043            this.indicesService = indicesService;
044        }
045    
046        @Override
047        protected String transportAction() {
048            return ReconstructIndexAction.NAME;
049        }
050    
051        @Override
052        protected String executor() {
053            return ThreadPool.Names.GET;
054        }
055    
056        @Override
057        protected ReconstructIndexRequest newRequest() {
058            return new ReconstructIndexRequest();
059        }
060    
061        @Override
062        protected ReconstructIndexResponse newResponse(ReconstructIndexRequest reconstructIndexRequest, AtomicReferenceArray shardsResponses, ClusterState clusterState) {
063            int successfulShards = 0;
064            int failedShards = 0;
065            List<ShardOperationFailedException> shardFailures = null;
066            final List<ShardReconstructIndexResponse> shards = newArrayList();
067            for (int i = 0; i < shardsResponses.length(); i++) {
068                Object shardResponse = shardsResponses.get(i);
069                if (shardResponse == null) {
070                    // a non active shard, ignore...
071                } else if (shardResponse instanceof BroadcastShardOperationFailedException) {
072                    failedShards++;
073                    if (shardFailures == null) {
074                        shardFailures = newArrayList();
075                    }
076                    shardFailures.add(new DefaultShardOperationFailedException((BroadcastShardOperationFailedException) shardResponse));
077                } else {
078                    shards.add((ShardReconstructIndexResponse)shardResponse);
079                    successfulShards++;
080                }
081            }
082            return new ReconstructIndexResponse(shards, shardsResponses.length(), successfulShards, failedShards, shardFailures);
083        }
084    
085        @Override
086        protected ShardReconstructIndexRequest newShardRequest() {
087            return new ShardReconstructIndexRequest();
088        }
089    
090        @Override
091        protected ShardReconstructIndexRequest newShardRequest(ShardRouting shardRouting, ReconstructIndexRequest reconstructIndexRequest) {
092            return new ShardReconstructIndexRequest(shardRouting.index(), shardRouting.id(), reconstructIndexRequest);
093        }
094    
095        @Override
096        protected BroadcastShardOperationResponse newShardResponse() {
097            return new ShardReconstructIndexResponse();
098        }
099    
100        @Override
101        protected ShardReconstructIndexResponse shardOperation(ShardReconstructIndexRequest request) throws ElasticsearchException {
102            IndexService indexService = indicesService.indexService(request.index());
103            InternalIndexShard indexShard = (InternalIndexShard) indexService.shardSafe(request.shardId());
104            Engine.Searcher searcher = indexShard.engine().acquireSearcher("transport_reconstruct");
105            IndexReader reader = searcher.reader();
106            DocumentReconstructor dr = new DocumentReconstructor(reader);
107            try {
108                return new ShardReconstructIndexResponse(true, dr.reconstruct(request.shardId()));
109            } catch (IOException e) {
110                throw new ElasticsearchException("failed to reconstruct index", e);
111            } finally {
112                searcher.release();
113            }
114        }
115    
116        @Override
117        protected GroupShardsIterator shards(ClusterState clusterState, ReconstructIndexRequest reconstructIndexRequest, String[] concreteIndices) {
118            return clusterState.routingTable().allActiveShardsGrouped(concreteIndices, true);
119        }
120    
121        @Override
122        protected ClusterBlockException checkGlobalBlock(ClusterState state, ReconstructIndexRequest request) {
123            return state.blocks().globalBlockedException(ClusterBlockLevel.METADATA);
124        }
125    
126        @Override
127        protected ClusterBlockException checkRequestBlock(ClusterState state, ReconstructIndexRequest reconstructIndexRequest, String[] concreteIndices) {
128            return state.blocks().indicesBlockedException(ClusterBlockLevel.METADATA, concreteIndices);
129        }
130    
131    }
