004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.support.master.MasterNodeOperationRequestBuilder;
006import org.elasticsearch.client.ClusterAdminClient;
007import org.elasticsearch.client.internal.InternalClusterAdminClient;
008
009/**
010 *  Consistency check request builder
011 *
012 */
013public class ConsistencyCheckRequestBuilder extends MasterNodeOperationRequestBuilder<ConsistencyCheckRequest, ConsistencyCheckResponse, ConsistencyCheckRequestBuilder> {
014
015    public ConsistencyCheckRequestBuilder(ClusterAdminClient clusterClient) {
016        super((InternalClusterAdminClient) clusterClient, new ConsistencyCheckRequest());
017    }
018
019    @Override
020    protected void doExecute(ActionListener<ConsistencyCheckResponse> listener) {
021        ((ClusterAdminClient) client).execute(ConsistencyCheckAction.INSTANCE, request, listener);
022    }
023}
