019package org.xbib.elasticsearch.action.skywalker;
020
021import org.elasticsearch.action.ShardOperationFailedException;
022import org.elasticsearch.action.support.broadcast.BroadcastOperationResponse;
023import org.elasticsearch.common.io.stream.StreamInput;
024import org.elasticsearch.common.io.stream.StreamOutput;
025
026import java.io.IOException;
027import java.util.HashMap;
028import java.util.List;
029import java.util.Map;
030
031/**
032 * A response for a skywalker action.
033 *
034 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
035 */
036public class SkywalkerResponse extends BroadcastOperationResponse {
037
038    private Map<String, Map<String, Map<String, Object>>> response;
039
040    SkywalkerResponse() {
041    }
042
043    SkywalkerResponse(int totalShards, int successfulShards, int failedShards, List<ShardOperationFailedException> shardFailures) {
044        super(totalShards, successfulShards, failedShards, shardFailures);
045    }
046
047    public SkywalkerResponse setResponse(Map<String, Map<String, Map<String, Object>>> response) {
048        this.response = response;
049        return this;
050    }
051
052    public Map<String, Map<String, Map<String, Object>>> getResponse() {
053        return response;
054    }
055
056    @Override
057    public void readFrom(StreamInput in) throws IOException {
058        super.readFrom(in);
059        int indexCount = in.readInt();
060        this.response = new HashMap();
061        for (int i = 0; i < indexCount; i++) {
062            String index = in.readString();
063            Map<String, Map<String, Object>> shards = new HashMap();
064            int shardCount = in.readInt();
065            for (int j = 0; j < shardCount; j++) {
066                String shard = in.readString();
067                Map<String, Object> fields = in.readMap();
068                shards.put(shard, fields);
069            }
070            response.put(index, shards);
071        }
072    }
073
074    @Override
075    public void writeTo(StreamOutput out) throws IOException {
076        super.writeTo(out);
077        int indexCount = response.keySet().size();
078        out.writeInt(indexCount);
079        for (String index : response.keySet()) {
080            out.writeString(index);
081            int shardCount = response.get(index).keySet().size();
082            out.writeInt(shardCount);
083            for (String shard : response.get(index).keySet()) {
084                out.writeString(shard);
085                out.writeMap(response.get(index).get(shard));
086            }
087        }
088    }
089}
