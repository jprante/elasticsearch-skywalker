019package org.xbib.elasticsearch.rest.action.skywalker;
020
021import org.elasticsearch.action.ActionListener;
022import org.elasticsearch.client.Client;
023import org.elasticsearch.common.inject.Inject;
024import org.elasticsearch.common.settings.Settings;
025import org.elasticsearch.common.xcontent.XContentBuilder;
026import org.elasticsearch.common.xcontent.XContentHelper;
027import org.elasticsearch.common.xcontent.XContentParser;
028import org.elasticsearch.rest.BaseRestHandler;
029import org.elasticsearch.rest.RestChannel;
030import org.elasticsearch.rest.RestController;
031import org.elasticsearch.rest.RestRequest;
032import org.elasticsearch.rest.XContentRestResponse;
033import org.elasticsearch.rest.XContentThrowableRestResponse;
034import org.elasticsearch.rest.action.support.RestXContentBuilder;
035import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexAction;
036import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexRequest;
037import org.xbib.elasticsearch.action.admin.indices.reconstruct.ReconstructIndexResponse;
038import org.xbib.elasticsearch.action.admin.indices.reconstruct.ShardReconstructIndexResponse;
039
040import java.io.IOException;
041
042import static org.elasticsearch.rest.RestRequest.Method.GET;
043import static org.elasticsearch.rest.RestStatus.OK;
044
045/**
046 *  REST reconstruct document action
047 *
048 *  @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
049 */
050public class RestReconstructIndexAction extends BaseRestHandler {
051
052    @Inject
053    public RestReconstructIndexAction(Settings settings, Client client, RestController controller) {
054        super(settings, client);
055        controller.registerHandler(GET, "/{index}/_reconstruct", this);
056    }
057
058    @Override
059    public void handleRequest(final RestRequest request, final RestChannel channel) {
060        client.admin().indices().execute(ReconstructIndexAction.INSTANCE,
061                new ReconstructIndexRequest(request.param("index")),
062                new ActionListener<ReconstructIndexResponse>() {
063                    @Override
064                    public void onResponse(ReconstructIndexResponse response) {
065                        try {
066                            XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
067                            builder.startObject()
068                                    .field("ok", true)
069                                    .field("index", request.param("index"))
070                                    .startArray("shards");
071                            for (ShardReconstructIndexResponse r : response.shards()) {
072                                XContentParser p = XContentHelper.createParser(r.getReconstructedIndex().bytes());
073                                builder.copyCurrentStructure(p);
074                            }
075                            builder.endArray();
076                            builder.endObject();
077                            channel.sendResponse(new XContentRestResponse(request, OK, builder));
078                        } catch (Exception e) {
079                            onFailure(e);
080                        }
081                    }
082
083                    @Override
084                    public void onFailure(Throwable e) {
085                        try {
086                            logger.error(e.getMessage(), e);
087                            channel.sendResponse(new XContentThrowableRestResponse(request, e));
088                        } catch (IOException e1) {
089                            logger.error("Failed to send failure response", e1);
090                        }
091                    }
092                });
093
094    }
095}
